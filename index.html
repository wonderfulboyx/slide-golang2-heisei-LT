	x, _ := strconv.Atoi(a)
	y, _ := strconv.Atoi(b)
	return x + y, nil
}

エラーハンドリングしようとすると…

func addStrAsInt(a, b string) (int, error) {
	x, err := strconv.Atoi(a)
	if err != nil {
		return 0, err
	}
	y, err := strconv.Atoi(b)
	if err != nil {
		return 0, err
	}
	return x + y, nil
}
